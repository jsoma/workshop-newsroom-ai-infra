PDF_DIR = Path("data/pdfs")

rows = []

for pdf_file in sorted(PDF_DIR.rglob("*.pdf")):
    with fitz.open(pdf_file) as pdf:
        for page_number, page in enumerate(pdf, start=1):
            text = " ".join(page.get_text().split())

            if text:
                rows.append({
                    "file": pdf_file.name,
                    "page": page_number,
                    "text": text,
                })

pages = pd.DataFrame(rows)

pages.head()
